'use strict';

import React, { Component } from 'react';
import { connect } from 'react-redux';
import {
	Text,
	View,
	Animated,
} from 'react-native';
import Button from 'aurochs/app/components/button';
// import Draggable from 'aurochs/app/components/draggable';
import styles from './Style';
/**
 * Main class
 */
class Main extends Component {
	/**
	 * The main view for the app
	 * @param props
	 * @param me
	 * @param store
	 */
	constructor(props, me, store) {
		super(props);
		this.food = () => {
			console.log('food');
			const age = this.props.age;
			this.props.dispatch({type: 'INCREMENT_AGE', age});
		}
		props.dispatch({type: 'INCREMENT_AGE', age: 20});
	}
	render () {
		const growstyle = () => {
			const a = this.props.age;
			const b = ("#" + ('000000' + this.props.counter.toString(16)).substr(-6, 6)).toString();
			return {
				height: a,
				width: a,
				borderRadius: a,
				backgroundColor: b
			}
		}
		return (
			<View style={styles.view}>
				<View style={growstyle()}></View>
				<Animated.View style={{width: 40, height: 40, backgroundColor: 'red'}}></Animated.View>
				<Text style={{fontSize: 20}}>{("#" + ('000000' + this.props.counter.toString(16)).substr(-6, 6)).toString()}</Text>
				<Text style={styles.text}>{Math.round(this.props.age)}</Text>
				<Text>{this.props.age}</Text>
				<Button label="hodang" />
				<Button style={styles.button} onPress={() => this.food()} label="Hobob" />
			</View>
		)
	}
}

const mapStateToProps = (state) => {

	}
}
export default connect(mapStateToProps)(Main)
